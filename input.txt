#include<stdio.h>
#include<stdbool.h>
int sum(int n);
int square(int n);
void foo(bool option);
int flag=0;

//main function
void main(){
    int num,result;
    printf("Enter a positive integer:\n");
    scanf("%d",&num);
    flag=-flag;
    foo(true);
    if(flag==1) 
      result=sum(num);
    else if(flag==0) 
      result=square(num);
    printf("result=%d\n",result);
}

/*sum function*/
int sum(int n){
    if(n==0)
       return n;
    else
       return n+sum(n-1);    
}

//square function
int square(int n){
    int i,j,result=0;
       for(i = 1; i <= n; i++){
          for(j=1;j<=n;j++){
             result=result+(1+4-2*2/1);
          }
       }
    return result;
}

//foo function
void foo(bool option){
    if(option==true)
       flag=1;
    else if(option==false)
       flag=0;
}
